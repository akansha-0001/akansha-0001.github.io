#shell script for factorial of a number
#factorial using while loop
echo "Enter a number"
read num
fact=1
while [ $num -gt 1 ]
do
  fact=$((fact * num))  #fact = fact * num
  num=$((num - 1))      #num = num - 1
done
echo $fact

# Program for Fibonacci
# Series
# Static input fo N
N=6
# First Number of the
# Fibonacci Series
a=0
# Second Number of the
# Fibonacci Series
b=1
echo "The Fibonacci series is : "
for (( i=0; i<N; i++ ))
do
	echo -n "$a "
	fn=$((a + b))
	a=$b
	b=$fn
done
